## 6) Post-run snapshots and DB diagnostics

curl -sS http://127.0.0.1:8080/actuator/prometheus
curl -sS http://127.0.0.1:9090/api/v1/targets
docker stats --no-stream modern-upo-sandbox-order-service modern-upo-sandbox-gateway modern-upo-sandbox-product-service
docker logs --since 20m modern-upo-sandbox-gateway > scripts/test/7-3/2026-04-24-7.3.9-gateway-logs-post.log 2>&1
docker logs --since 20m modern-upo-sandbox-order-service > scripts/test/7-3/2026-04-24-7.3.9-order-service-logs-post.log 2>&1
docker logs --since 20m modern-upo-sandbox-product-service > scripts/test/7-3/2026-04-24-7.3.9-product-service-logs-post.log 2>&1
docker exec modern-upo-sandbox-mysql mysql -uroot -proot_pass -N -e "SELECT id,product_code,stock,status FROM h_product_db.t_product WHERE id=2;"
docker exec modern-upo-sandbox-mysql mysql -uroot -proot_pass -N -e "SELECT status, COUNT(*) FROM h_order_db.t_order GROUP BY status ORDER BY status;"
docker exec modern-upo-sandbox-mysql mysql -uroot -proot_pass -N -e "SELECT user_id, request_no, COUNT(*) cnt FROM h_order_db.t_order WHERE request_no IS NOT NULL GROUP BY user_id, request_no HAVING cnt > 1;"
docker exec modern-upo-sandbox-mysql mysql -uroot -proot_pass -N -e "SELECT status, COUNT(*) FROM h_order_db.t_order_coupon_issue_task GROUP BY status ORDER BY status;"
docker exec modern-upo-sandbox-mysql mysql -uroot -proot_pass -N -e "SELECT COUNT(*) AS due_shipping FROM h_order_db.t_order WHERE status = 6 AND expected_delivery_time <= NOW();"
docker exec modern-upo-sandbox-mysql mysql -uroot -proot_pass -N -e "SELECT waiting_pid, waiting_query, blocking_pid, blocking_query, wait_age_secs, locked_table, locked_index, waiting_lock_mode, blocking_lock_mode FROM sys.innodb_lock_waits;"
docker exec modern-upo-sandbox-mysql mysql -uroot -proot_pass -N -e "SELECT SCHEMA_NAME, DIGEST_TEXT, COUNT_STAR, ROUND(SUM_TIMER_WAIT/1000000000000,3) AS SUM_WAIT_S, ROUND(AVG_TIMER_WAIT/1000000000,3) AS AVG_WAIT_MS, ROUND(SUM_LOCK_TIME/1000000000000,3) AS LOCK_S FROM performance_schema.events_statements_summary_by_digest ORDER BY SUM_TIMER_WAIT DESC LIMIT 30;"
